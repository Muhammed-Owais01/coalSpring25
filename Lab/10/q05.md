```asm
INCLUDE Irvine32.inc

.data
    nums DWORD 5, 2, 8, 3, 1

.code

BubbleSort PROC
    push ebp
    mov ebp, esp

    mov ecx, [ebp + 12]
    dec ecx
Outer_Loop:
    push ecx
    mov esi, [ebp + 8]
    mov ecx, [ebp + 12]
    dec ecx
Inner_Loop:
    mov eax, DWORD PTR [esi]
    mov edx, DWORD PTR [esi + 4]
    cmp eax, edx
    jle _LOOP

    mov [esi], edx
    mov [esi + 4], eax
_LOOP:
    add esi, TYPE DWORD
    loop Inner_Loop

    pop ecx
    loop Outer_Loop

    pop ebp
    ret 8
BubbleSort ENDP

main PROC
    mov ecx, LENGTHOF nums
    mov esi, 0
L1:
    mov eax, nums[esi * TYPE nums]
    call WriteDec
    mov eax, 32
    call WriteChar

    inc esi
    loop L1
    
    push OFFSET nums
    push LENGTHOF nums
    call BubbleSort
    
    call Crlf
    mov ecx, LENGTHOF nums
    mov esi, 0
L2:
    mov eax, nums[esi * TYPE nums]
    call WriteDec
    mov eax, 32
    call WriteChar

    inc esi
    loop L2
    exit
main ENDP
END main
```

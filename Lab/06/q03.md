```asm
INCLUDE Irvine32.inc

.data
	numEmployees EQU 5

	empId DWORD 5 DUP(?)
	empName BYTE 5 * 30 DUP(?)
	birth DWORD 5 DUP(?)
	salary DWORD 5 DUP(?)

	totalSalary DWORD 0

	promptId BYTE "Enter Employee ID: ", 0
	promptName BYTE "Enter Employee Name: ", 0
	promptBirth BYTE "Enter Year Of Birth: ", 0
	promptSalary BYTE "Enter Annual Salary: ", 0
	outputTotal BYTE "Total Annual Salary: ", 0

.code
main proc
	mov esi, 0

	input_loop:
		mov edx, offset promptId
		call WriteString
		call ReadInt
		mov empId[esi * type empId], eax

		mov edx, offset promptName
		call WriteString
		mov edx, offset empName
		mov eax, esi
		imul eax, 30  
		add edx, eax
		mov ecx, 30
		call ReadString

		mov edx, offset promptBirth
		call WriteString
		call ReadInt
		mov birth[esi * type birth], eax

		mov edx, offset promptSalary
		call WriteString
		call ReadInt
		mov salary[esi * type salary], eax

		inc esi
		cmp esi, numEmployees
		jl input_loop

		mov esi, 0
		mov totalSalary, 0

	sum_salary:
		mov eax, totalSalary
		add eax, salary[esi * type salary]
		mov totalSalary, eax

		inc esi
		cmp esi, numEmployees
		jl sum_salary

		mov edx, OFFSET outputTotal
		call WriteString
		mov eax, totalSalary
		call WriteDec
		call CrLf
exit
main endp
end main
```

```asm
INCLUDE Irvine32.inc

	
.data
	val1 BYTE 10h
	val2 WORD 8000h
	val3 DWORD 0FFFFh
	val4 WORD 7FFFh
.code
main proc
	; Write an instruction that increments val2.
	mov ax, val2
	inc ax
	call writeint
	call crlf
	
	; Write an instruction that subtracts val3 from EAX.
	mov eax, 5000h
	sub eax, val3
	call writeint
	call crlf

	; Write instructions that subtract val4 from val2.
	mov ax, val2
	sub ax, val4
	call writeint
exit
main ENDP
END main
```
